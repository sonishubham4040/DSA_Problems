import java.util.*;

public class Solution
{
    public static void main(String[] args)
    {
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
        Student[] students=new Student[n];
        
        for(int i=0;i<n;i++){
            int id=sc.nextInt();
            sc.nextLine();
            String name=sc.nextLine();
            int marks=sc.nextInt();
            students[i]=new Student(id, name, marks);
        }
        
        String ht=highTotal(students);
        System.out.println(ht);
          
        int[] ans=searchPercentage(students);
        if(ans!=null){
            for(int i=0;i<ans.length();i++){
                System.out.println(ans[i]);
            }
        }
        else{
            System.out.println("No student found with mentioned attribute.");
        }
        
        sc.close();
    }
    
    public static String highTotal(Student[] students){
        Arrays.sort(students, Comparator.comparing(Student::getMarks).reversed);
        
        return students[0].getName().toUpperCase();
    }
    public static int[] searchPercentage(Student[] students){
        List<Integer> ids=new ArrayList<>();
        
        for(Student s: students){
            if(s.getMarks()/4>70){
                ids.add(s.getId());
            }
        }
        if(ids.isEmpty()){
            return null;
        }
        
        Collections.sort(ids);
        // return type 1
        // int[] arr=new int[ids.size()];
        // for(int i=0;i<ids.size();i++){
        //     arr[i]=ids.get(i);
        // }
        // return arr;
        
        // return type 2
        return ids.stream().mapToInt(Integer::intValue).toArray();
    }
}

{
    private int id;
    private String name;
    private int marks;
    
    public Student(int id, String name, int marks){
        this.id=id;
        this.name=name;
        this.marks=marks;
    } 
    
    //get
    public int getId(){
        return id;
    }
    //.....
    
    //set
    public void setId(int id){
        this.id=id;
    }
    // .......
}

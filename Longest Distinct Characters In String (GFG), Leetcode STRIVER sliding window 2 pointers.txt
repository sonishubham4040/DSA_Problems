			// Longest Distinct characters in string (GFG), Leetcode-3
			// STRIVER 2 pointers, sliding window

int longestSubstrDistinctChars (string S)
{
    int hash[256];
    fill(hash, hash+256, -1);
    
    int n=S.size();
    int l=0, r=0, maxi=0;
    
    while(r<n)
    {
        if(hash[S[r]]!=-1 && hash[S[r]]>=l)
        {
            l=hash[S[r]]+1;
        }
        
        maxi=max(maxi, r-l+1);
        hash[S[r]]=r;
        r++;
    }
    
    return maxi;
}

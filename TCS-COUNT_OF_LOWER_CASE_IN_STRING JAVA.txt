		// COUNT OF LOWER CASE IN STRING
		// TCS - 3 APPROACHES

// APPROACH 1
import java.util.*;

class Solution
{
    public static void main(String[] args)
    {
        Scanner sc=new Scanner(System.in);
        String s=sc.nextLine();
        int cnt=0;
        
        for(int i=0;i<s.length();i++){
            char ch=s.charAt(i);
            if(Character.isLowerCase(ch)){
                cnt++;
            }    
        }
        
        System.out.println("lower count is: "+cnt);
        sc.close();
    }
}

import java.util.*;

class Solution
{
    public static void main(String[] args)
    {
        Scanner sc=new Scanner(System.in);
        String s=sc.nextLine();
        int cnt=0;
        
        for(int i=0;i<s.length();i++){
            char ch=s.charAt(i);
            if(ch>='a' && ch<='z'){
                cnt++;
            }    
        }
        
        System.out.println("lower count is: "+cnt);
        sc.close();
    }
}

import java.util.*;

class Solution
{
    public static void main(String[] args)
    {
        Scanner sc=new Scanner(System.in);
        String s=sc.nextLine();
        int cnt=0;
        
        for(char ch: s.toCharArray()){
            if(Character.isLowerCase(ch)){
                cnt++;
            }    
        }
        
        System.out.println("lower count is: "+cnt);
        sc.close();
    }
}
